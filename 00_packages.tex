\usepackage{mdframed}
\newcounter{box}[section]\setcounter{box}{0}
\renewcommand{\thebox}{\arabic{section}\textbar\arabic{box}}

%======================
% === example box ===
% =====================
\newcounter{examp}[section]\setcounter{examp}{0}
\renewcommand{\theexamp}{\arabic{section}.\arabic{examp}}
\newenvironment{examp}[2][]{%
    % \refstepcounter{examp}%
    \refstepcounter{box}%
    \ifstrempty{#1}%
    {\mdfsetup{%
            frametitle={%
                    \tikz[baseline=(current bounding box.east),outer sep=0pt]
                    \node[anchor=east,rectangle,fill=blue!20]
                    {\strut Example~\thebox};}}
    }%
    {\mdfsetup{%
            frametitle={%
                    \tikz[baseline=(current bounding box.east),outer sep=0pt]
                    \node[anchor=east,rectangle,fill=blue!20]
                    {\strut Example~\thebox:~#1};}}%
    }%
    \mdfsetup{innertopmargin=10pt,linecolor=blue!20,%
        linewidth=2pt,topline=true,%
        frametitleaboveskip=\dimexpr-\ht\strutbox\relax
    }
    \begin{mdframed}[]\relax%
        \label{#2}}{\end{mdframed}}
